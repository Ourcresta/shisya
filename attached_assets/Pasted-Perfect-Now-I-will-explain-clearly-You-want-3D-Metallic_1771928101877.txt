# ‚úÖ 3Ô∏è‚É£ main.js (FULL PREMIUM VERSION)

üî• This is the real magic

```javascript
const canvas = document.getElementById("coinCanvas");

// Scene
const scene = new THREE.Scene();

// Camera
const camera = new THREE.PerspectiveCamera(
  50,
  window.innerWidth / window.innerHeight,
  0.1,
  1000
);
camera.position.z = 5;

// Renderer
const renderer = new THREE.WebGLRenderer({ canvas, antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.setPixelRatio(window.devicePixelRatio);

// Lights
const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);
scene.add(ambientLight);

const directionalLight = new THREE.DirectionalLight(0xffffff, 1);
directionalLight.position.set(5, 5, 5);
scene.add(directionalLight);

// Geometry (Coin Shape)
const geometry = new THREE.CylinderGeometry(2, 2, 0.3, 100);

// Load Texture
const textureLoader = new THREE.TextureLoader();
const coinTexture = textureLoader.load("coin.png");

// Materials
const sideMaterial = new THREE.MeshStandardMaterial({
  color: 0xc0c0c0,
  metalness: 1,
  roughness: 0.3,
});

const frontMaterial = new THREE.MeshStandardMaterial({
  map: coinTexture,
  metalness: 0.8,
  roughness: 0.4,
});

const backMaterial = new THREE.MeshStandardMaterial({
  map: coinTexture,
  metalness: 0.8,
  roughness: 0.4,
});

const materials = [sideMaterial, frontMaterial, backMaterial];

// Mesh
const coin = new THREE.Mesh(geometry, materials);
scene.add(coin);

// Initial tilt
coin.rotation.x = 0.3;

// Floating animation
let floatDirection = 1;

// Mouse interaction
window.addEventListener("mousemove", (event) => {
  const mouseX = (event.clientX / window.innerWidth) - 0.5;
  const mouseY = (event.clientY / window.innerHeight) - 0.5;

  coin.rotation.y += mouseX * 0.05;
  coin.rotation.x += mouseY * 0.05;
});

// Click spin
window.addEventListener("click", () => {
  gsap.to(coin.rotation, {
    y: coin.rotation.y + Math.PI * 2,
    duration: 1,
    ease: "power2.out"
  });
});

// Animation Loop
function animate() {
  requestAnimationFrame(animate);

  // Slow rotation
  coin.rotation.y += 0.01;

  // Floating
  coin.position.y += 0.005 * floatDirection;

  if (coin.position.y > 0.3) floatDirection = -1;
  if (coin.position.y < -0.3) floatDirection = 1;

  renderer.render(scene, camera);
}

animate();

// Resize Fix
window.addEventListener("resize", () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});
```
